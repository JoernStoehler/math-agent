\begin{exercise} % Aufgabe 5
  Sei $I$ ein Intervall und seien $f,g \colon I \to \R$ zwei konvexe Funktionen
  und $\alpha > 0$. Zeige dann, dass
  \begin{enumerate}[label=(\alph*)]
  \item $\alpha \cdot f$
  \item $f + g$
  \item $\max\set{f,g}$
  \end{enumerate}
  ebenfalls konvex sind.
\end{exercise}
