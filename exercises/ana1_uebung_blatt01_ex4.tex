\begin{exercise}
  Leite die folgenden Gleichheiten aus den Axiomen der Vorlesung her ohne
  Wahrheitstafeln zu verwenden.
  \begin{enumerate}
  \item $A \lor A = A$
  \item $A \land A = A$
  \item $A \lor 1 = 1$
  \item $A \land 1 = A$
  \item $A \lor 0 = A$
  \item $A \land 0 = 0$
  \end{enumerate}
\end{exercise}
