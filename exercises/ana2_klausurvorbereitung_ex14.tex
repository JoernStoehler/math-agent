        \begin{split}
            f:\R^2&\to\R\\
            (x,y)&\mapsto \begin{cases}
						x^3y^2\sin\left(\frac{1}{x^2+y}\right),& x^2+y\neq 0\\
						0,& x^2+y=0
						\end{cases}
        \end{split}
    \end{equation*}

\end{prob}
