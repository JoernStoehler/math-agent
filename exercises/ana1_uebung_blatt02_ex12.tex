\begin{exercise}[Injektivität, Surjektivität]
  Welche der folgenden Funktionen sind injektiv, welche surjektiv, und
  welche bijektiv?
  \begin{enumerate}
  \item
    \begin{center}
      \begin{tikzcd}[row sep=tiny]
        \mathbb{Z}
        \arrow [r, "f"]
        & \mathbb{Z}
        & x
        \arrow [r, mapsto]
        &	\abs{ x } \\
      \end{tikzcd}
    \end{center}

  \item
    \begin{center}
      \begin{tikzcd}[row sep=tiny]
        \mathbb{R}
        \arrow [r, "f"]
        & \mathbb{R}_{0}^{+}
        & x
        \arrow [r, mapsto]
        & x^2 \\
      \end{tikzcd}
    \end{center}
  \item
    \begin{center}
      \begin{tikzcd}[row sep=tiny]
        \mathbb{Z}
        \arrow [r, "f"]
        & \mathbb{N}
        & x
        \arrow [r, mapsto]
        & \abs{ x } \\
      \end{tikzcd}
    \end{center}
  \item
    \begin{center}
      \begin{tikzcd}[row sep=tiny]
        \mathbb{Z}
        \arrow [r, "f"]
        & \mathbb{N} 
        & x
        \arrow [r, mapsto]
        & x^2 \\
      \end{tikzcd}
    \end{center}
  \end{enumerate}
\end{exercise}
