            \begin{prob}[Riemann- und Lebesgue-Integrierbarkeit] \\
%
Für welche $s \in \R$ ist die Funktion
$$
f \colon ( 0 , \i) \mapsto \R \colon x \mapsto x^s \sin x
$$
\begin{enumerate}[label = (\alph*)]
	\item uneigentlich Riemann-integrierbar bzgl.~des aussch\"opfenden
Systems $\mathcal{K} = \{[a,b]\mid 0<a<b<\infty\}$?

	\item Lebesgue-integrierbar?
\end{enumerate}
%------------------------------------------------------------------------------------------------------------------------
\vspace{2mm}
            \end{prob}
