\begin{prob}[Einige Integrale]
%
Berechnen Sie $\int_M f \, d \lambda$ für
\begin{enumerate}[label = (\alph*)]
	\item $M = \left\{ \left( x , y , z \right) \in \R^3 \mid \left( x , y \right) A \left( x , y \right)^\top \le z \le 1 \right\}$ für eine positiv definite Matrix $A \in \R^{2 \times 2}_{sym}$ und $f(x, y, z) \equiv 1$.
	\item $M = \left\{ \left( x , y , z \right) \in \R^3 \mid 0 \le x \le y, x^2 + y^2 \le 4, 0 \le z \le x y \right\}$ und $f(x, y, z) = x^2 + y^2$.
	\item $M = \left\{ \left( x , y \right) \in \R^2 \mid 0 \le x - y \le 1, 0 \le x + y \le 1 \right\}$ und $f(x, y) = \frac{1}{1 - x^2 + y^2}$.
\end{enumerate}
%------------------------------------------------------------------------------------------------------------------------
\vspace{2mm}
\end{prob}
