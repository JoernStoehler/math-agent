\begin{prob}
Berechne die Jacobimatrix der Funktion 
    \begin{equation*}
        \begin{split}
            f:\R^4&\to\R^2\\
            (w,x,y,z)&\mapsto 
            \begin{pmatrix}
            \sin(w)\sin(z)\\
            e^{x-y}
            \end{pmatrix}
        \end{split}
    \end{equation*}
    und bestimme ihren Rang in Abh"angigkeit von $(w,x,y,z)$.
\end{prob}
