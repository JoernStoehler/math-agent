                    \begin{prob}[\bf Eigenschaften Jordan-messbarer Mengen]
%
Zeigen Sie:
\begin{enumerate}[label = (\alph*)]
	\item Für $A \in \mathcal{J}(\R^n)$ gilt $\mathfrak{J}(A) = \lim_{k \to \i} k^{-n} \left| A \cap k^{-1} \Z^n \right|$. \label{Gitter-Limes-Bedingung}
	\item Zeigen Sie, dass \ref{Gitter-Limes-Bedingung} nicht für jede beschränkte Borelmenge $A$ gilt.
\end{enumerate}
$Tipp$: Beweisen Sie $\ref{Gitter-Limes-Bedingung}$ zunächst für Quader.
%------------------------------------------------------------------------------------------------------------------------
\vspace{2mm}
                    \end{prob}
