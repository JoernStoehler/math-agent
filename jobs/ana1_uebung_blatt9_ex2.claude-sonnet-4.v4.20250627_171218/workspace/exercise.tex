\begin{prob}
  Beweise für alle $n \in \N$ und alle $x \in \R$ die beiden folgenden
  Identitäten:
  \begin{equation*}
    \begin{split}
      \cos(nx) &= \sum_{k=0}^{\floor{\frac{n}{2}}} (-1)^k\binom{n}{2k} \cos(x)^{n-2k}\sin(x)^{2k}\\
      \sin(nx) &= \sum_{k=0}^{\floor{\frac{n}{2}}} (-1)^k\binom{n}{2k+1} \cos(x)^{n-2k-1}\sin(x)^{2k+1}
    \end{split}
  \end{equation*}
\end{prob}
