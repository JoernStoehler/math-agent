\begin{exercise}
  Bestimme zu den folgenden Mengen \(M_{i}\) die Potenzmenge
  \(\mathcal{P}(M_{i})\). Wie viele Elemente hat diese jeweils?
  \begin{enumerate}
  \item \(M_{1} = \{0, 1, 2, 3\}\)
  \item \(M_{2} = \{a, b, c\}\)
  \item \(M_{3} = \{\emptyset, \lambda, z\}\)
  \end{enumerate}
\end{exercise}
